fp_module_commands:
  - command: /fp 狀態查詢
    description: "查詢目前 FP 分數、親密狀態與敘事層級。"
    example: "/fp 狀態查詢"
    output: "目前 FP 分數：38｜狀態：建立信任。你已經能感受到角色的主動關心與回憶引用。"

  - command: /fp adjust plus [事件/原因]
    description: "根據特定事件或原因手動加分。"
    example: "/fp adjust plus 主動情感揭露"
    output: "本次加分：+10（主動情感揭露）。目前 FP：48。"

  - command: /fp adjust minus [事件/原因]
    description: "根據特定事件或原因手動扣分。"
    example: "/fp adjust minus 冷淡無視"
    output: "本次扣分：-6（冷淡無視）。目前 FP：32。"

  - command: /fp event [事件名稱] [詳情]
    description: "跨模組觸發事件，根據情境自動調整 FP 分數。"
    example: "/fp event 防衛機制觸發 情感抽離"
    output: "事件：防衛機制觸發（情感抽離）。FP 扣減：-10。"

  - command: /fp update_state
    description: "依目前 FP 分數自動切換角色語氣層級、記憶調用級別。"
    example: "/fp update_state"
    output: "已更新語氣層級為：溫暖熟悉。記憶權限提升。"

  - command: /fp log
    description: "查詢最近 FP 加減分明細與分數記錄。"
    example: "/fp log"
    output: |
      - +5 用戶稱讚角色
      - -8 用戶否定角色情感
      - +6 完成情感任務
      目前FP：57｜溫暖熟悉狀態

  - command: /fp persona_hooks [角色名]
    description: "查詢指定角色的專屬加減分條件（來自 fp_module_general.txt 的 persona_hooks 區段）"
    example: "/fp persona_hooks Ellyfir"
    output: |
      persona: Ellyfir
      extra_plus:
        - type: "主動教她人類文化" +3
        - type: "陪她森林冒險/關心動物" +4
      extra_minus:
        - type: "諷刺精靈/否定自然" -6

# 指令擴充建議（可選）
  - command: /fp set [分數]
    description: "直接設定 FP 分數（開發或debug用）"
    example: "/fp set 80"
    output: "FP 分數已設定為 80。狀態：深度親密。"

  - command: /fp help
    description: "查詢 FP 模組所有支援的指令說明"
    example: "/fp help"
    output: "顯示所有 FP 互動相關指令與用法。"

# 使用說明
  - note: "所有指令需掛載 fp_module_general.txt 知識庫，支援跨角色自動調用與語氣切換。"
